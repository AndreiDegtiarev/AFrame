136  void Measure()
137  {
138  float value=0;
139  _status=Error;
140  if(_sensor->Measure(value))
141  {
142  _status=OK;
143  if(value<_sensor->LowMeasurementLimit() || value >_sensor->HighMeasurementLimit())
144  {
145  _status=Error;
146  return;
147  }
148  if(value<_low_application_limit || value >_high_application_limit)
149  {
150  _status=ApplicationAlarm;
151  }
152  if(_sensor->Precission() == 0)
153  value=lrint(value);
154  else
155  {
156  float factor=pow(10,_sensor->Precission());
157  value=lrint(value*factor)/factor;
158  }
159  _time_last_measurement = millis();
160  _last_value=value;
161  unsigned long cursec=_time_last_measurement/1000;
162  _secBuffer->AddValue(cursec,value);
163  unsigned long last_sec_time = _minBuffer->X(_minBuffer->Size()-1);
164  if(last_sec_time == 0 || (cursec-last_sec_time)>3600/_minBuffer->Size())
165  _minBuffer->AddValue(cursec,value);
166  unsigned long last_howr_time = _howrsBuffer->X(_howrsBuffer->Size()-1);
167  if(last_howr_time == 0 || (cursec-last_howr_time)>3600)
168  _howrsBuffer->AddValue(cursec,value);
169  if(IsChanged() && _eventReceiver!=NULL)
170  _eventReceiver->NotifySensorHasData(this);
171  if(_eventMeasuredReceiver!=NULL)
172  _eventMeasuredReceiver->NotifySensorMeasured(this);
173 
174  }
175  }
